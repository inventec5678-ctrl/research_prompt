## 回測框架（直接複製使用）

```python
#!/usr/bin/env python3
"""
[你的策略名稱]
"""
import sys
sys.path.insert(0, '/Users/changrunlin/.openclaw/workspace/crypto-agent-platform/autoresearch')

import numpy as np
import pandas as pd
from backtest_core import BacktestEngine, load_data, compute_indicators

# ============ 載入數據 ============
df = load_data()  # 載入 15分K
df = compute_indicators(df)

# ============ 你的策略 ============
def generate_signals(data, params):
    """
    生成交易信號
    
    Args:
        data: DataFrame（包含 close, high, low, volume, rsi, adx, atr, ma20, ma50, vol_ma20, cci）
        params: 參數字典
    
    Returns:
        signals: np.array（1=做多, -1=做空, 0=無信號）
    """
    signals = np.zeros(len(data))
    closes = data['close'].values
    rsi = data['rsi'].values
    adx = data['adx'].values
    atr = data['atr'].values
    volume = data['volume'].values
    vol_ma = data['vol_ma20'].values
    cci = data['cci'].values
    
    for i in range(50, len(data) - 1):
        # === 在這裡實現你的進場邏輯 ===
        
        # 範例：RSI 超賣時做多
        if rsi[i] < 30:
            signals[i] = 1
        elif rsi[i] > 70:
            signals[i] = -1
        
    return signals

# ============ 回測參數 ============
SL_PCT = 0.02      # 止損 2%
TP_PCT = 0.04      # 止盈 4%
MAX_BARS = 20      # 最大持倉 K 線數

# Trailing Stop 參數（可選）
USE_TRAILING = False
TRAIL_PCT = 0.02
ACTIVATION_PCT = 0.01

# ============ 測試函數 ============
def test_strategy(signals, params, sl_pct, tp_pct, max_bars, use_trailing=False, trail_pct=0.02, activation_pct=0.01):
    engine = BacktestEngine(df)
    result = engine.run(
        signals, 
        sl_pct=sl_pct, 
        tp_pct=tp_pct, 
        max_bars=max_bars,
        use_trailing_stop=use_trailing,
        trail_pct=trail_pct,
        activation_pct=activation_pct
    )
    return result

def print_result(result, label=""):
    print(f"\n=== {label} ===")
    print(f"WR: {result['wr']:.2f}%")
    print(f"PF: {result['pf']:.2f}")
    print(f"DD: {result['dd']:.2f}%")
    print(f"Sharpe: {result['sharpe']:.2f}")
    print(f"月交易: {result['monthly_trades']:.1f}")
    print(f"Exit: {result['exit_stats']}")
    print(f"通過: {result['passed']}")

# ============ 主程序 ============
if __name__ == "__main__":
    params = {}  # 你的參數
    
    # 生成信號
    signals = generate_signals(df, params)
    
    # 測試
    result = test_strategy(signals, params, SL_PCT, TP_PCT, MAX_BARS, USE_TRAILING, TRAIL_PCT, ACTIVATION_PCT)
    print_result(result, "基礎測試")
```

## 分析 → 修正流程

### 核心原則
**不要猜測問題，而是根據實際結果調整**

### 步驟 1：運行基礎測試

運行你的策略，觀察：
- WR（勝率）
- PF（盈虧比）
- Exit 分佈（SL/TP/Timeout/Trail）

### 步驟 2：分析 Exit 分佈

**看到什麼問題，就修什麼問題**

| Exit 分佈顯示 | 問題是 | 這樣調整 |
|---------------|--------|----------|
| Timeout 很高（>60%） | 交易太久了 | 縮短 max_bars 或開啟 Trailing Stop |
| Stop Loss 很高（>40%） | 止損被掃太多 | 放寬 SL 或加入波動過濾 |
| Take Profit 很低（<5%） | 價格不到止盈就回調 | 放寬 TP 或用 Trailing Stop |
| Trailing Stop 很多 | 市場來回觸發止損 | 收緊 TRAIL_PCT 或提高 ACTIVATION_PCT |

### 步驟 3：觀察 WR 和 PF

| 指標問題 | 可能原因 | 這樣調整 |
|---------|---------|----------|
| WR < 55% | 進場時機不好 | 加入過濾條件（ADX/RSI/CCI/Volume） |
| PF < 1.5 | 虧的太多或賺的太少 | 放寬 TP、縮窄 SL、或用 Trailing Stop |
| DD > 25% | 連續虧損 | 減少倉位或放寬 SL |
| Sharpe < 1.5 | 風險調整後收益差 | 減少交易頻率或加更嚴格過濾 |
| 月交易太少（<100） | 信號太少 | 放寬進場條件 |

### 步驟 4：應用修正

## 常用過濾條件

```python
# ADX 趨勢過濾
if adx[i] < 20:
    continue

# RSI 範圍過濾
if rsi[i] < 25 or rsi[i] > 75:
    continue

# 成交量確認
if volume[i] < vol_ma[i] * 1.2:
    continue

# ATR 波動過濾
if atr[i] / closes[i] < 0.005:
    continue

# CCI 過濾
if cci[i] > -100 or cci[i] < 100:
    continue
```
